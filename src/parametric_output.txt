package 'ParametricDiagrams' {
	part def Constraint{
		attribute equation : String;
		part def Variable;
	}
	part def Parameter;
	part def Connection{
		attribute variable_type : String;
	}

	part C3 : Constraint {
		attribute equation = "<<d>>";
		part d : Variable;
	}

	part C1 : Constraint {
		attribute equation = "a,b";
		part a : Variable;
		part b : Variable;
	}

	part C2 : Constraint {
		attribute equation = "a+c-d";
		part d : Variable;
		part a : Variable;
		part c : Variable;
	}

	part c : Parameter;
	part b : Parameter;

	part b : Connection {
		attribute variable_type = "int";
		connect Parameter.b to Constraint.C1.b;
	}
	part a : Connection {
		attribute variable_type = "int";
		connect Constraint.C2.a to Constraint.C1.a;
	}
	part c : Connection {
		attribute variable_type = "int";
		connect Parameter.c to Constraint.C2.c;
	}
	part d : Connection {
		attribute variable_type = "int";
		connect Constraint.C3.d to Constraint.C2.d;
	}
}
